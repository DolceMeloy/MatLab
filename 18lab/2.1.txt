% Определяем функцию для минимизации
f = @(x) -cos(x(1)) .* cos(x(2)) .* exp(-((x(1) - pi).^2 + (x(2) - pi).^2));

% Начальная точка
x0 = [0, 0];  % можно начать с (0,0), так как это область близка к минимуму

% Минимизация с использованием метода Нелдера-Мида
[x_min, fval] = fminsearch(f, x0);

% Отображаем результаты
disp('Минимум функции (x1, x2):');
disp(x_min);
disp('Значение функции в минимуме:');
disp(fval);
