% Определение функции Изома
f = @(x) -cos(x(1)) .* cos(x(2)) .* exp(-((x(1) - pi).^2 + (x(2) - pi).^2));

% Ограничения
nonlcon = @(x) deal([3*(x(1) - 2)^2 - (x(2) - 2)^3; x(1)^2 + x(2)^2 - 10], []);  % неравенства

% Начальная точка
x0 = [0, 0];

% Поиск минимума с использованием patternsearch
[x_opt, fval] = patternsearch(f, x0, [], [], [], [], [], [], nonlcon);

disp('Оптимальное значение (x1, x2):');
disp(x_opt);
disp('Значение функции в минимуме:');
disp(fval);
