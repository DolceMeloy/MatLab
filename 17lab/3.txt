% Коэффициенты целевой функции (максимизация)
f = [-6; -1];  % Так как MATLAB решает задачу минимизации, мы берем отрицательные значения

% Коэффициенты ограничений
A = [2, 7; 3, -8; -3, 3];
b = [27; 5; 4];

% Ограничения на переменные
lb = [0; 0];  % x1 >= 0, x2 >= 0
ub = [];  % Нет верхних ограничений для переменных

% Указание переменной x2 как целочисленной
intcon = 2;  % x2 — целое число (индекс 2)

% Начальная точка
x0 = [0; 0];

% Решение задачи линейного программирования с целочисленным ограничением
options = optimoptions('intlinprog', 'Display', 'iter');
[x, fval] = intlinprog(f, intcon, A, b, [], [], lb, ub, options);

% Печать результатов
disp('Оптимальные значения переменных:');
disp(x);

disp('Значение целевой функции в минимуме:');
disp(-fval);  % Так как мы минимизируем, возвращаем знак функции.
