% Коэффициенты функции цели
f = [4; 6];  % Стоимость каждого набора удобрений

% Ограничения: азотные, калийные, медные удобрения
A = [-3 -1; -1 -2; -1 -6];  % Мы используем отрицательные значения, потому что linprog решает задачу минимизации, а ограничения >=
b = [-9; -8; -12];  % Минимальные потребности в удобрениях (азотные, калийные и медные)

% Нижняя граница (удобрения >= 0)
lb = zeros(2, 1);  % Неотрицательные значения для количества наборов

% Решение задачи линейного программирования
[x, fval] = linprog(f, A, b, [], [], lb);

disp('Оптимальные объемы удобрений (Купрум-I, Купрум-II):');
disp(x);
disp('Минимальная стоимость:');
disp(fval);
