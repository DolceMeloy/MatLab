% Коэффициенты функции цели
H = [2, 0; 0, 2];  % матрица коэффициентов при квадратичных членах
f = [-18, 20];  % линейная часть функции

% Ограничения
A = [4, 3; -1, 4];  % коэффициенты при переменных для ограничений
b = [8; 4];  % правые части ограничений
lb = [0; -Inf];  % нижняя граница для x1

% Решение задачи квадратичного программирования
[x, fval] = quadprog(H, f, A, b, [], [], lb);

% Вывод результатов
disp('Оптимальные значения переменных:');
disp(x);
disp('Минимальное значение функции:');
disp(fval);
